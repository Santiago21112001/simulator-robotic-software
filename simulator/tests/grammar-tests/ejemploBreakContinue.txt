int sens;

void setup(){
    sens = 0;
}


void loop(){
    int threshold = 40;
    for (int x = 0; x < 255; x++) {
        analogWrite(PWMpin, x);
        sens = analogRead(sensorPin);
        if (sens > threshold) {     // bail out on sensor detect
            x = 0;
            break;
        }
        delay(50);
        continue;
    }
}
